<script type="module">

import { auth, db } from "./js/firebase.js";
import { onAuthStateChanged, signOut } from "https://www.gstatic.com/firebasejs/12.15.0/firebase-auth.js";
import { doc, getDoc, collection, getDocs, updateDoc, addDoc, arrayUnion } from "https://www.gstatic.com/firebasejs/12.15.0/firebase-firestore.js";

let candidatures = [];   // liste brute { id, ...data }
let filtreActuel = "toutes";
let candidatureOuverteId = null;

const LIBELLES_STATUT = {
    soumise: "🟠 En vérification",
    en_verification: "🟠 En vérification",
    documents_valides: "🟢 Documents validés",
    test_autorise: "🟣 Test autorisé",
    admis: "✅ Admis",
    refuse: "❌ Refusé"
};

function identifiantCourt(id) {
    return "CAND-" + id.slice(0, 6).toUpperCase();
}

function formaterDate(iso) {
    if (!iso) return "-";
    try { return new Date(iso).toLocaleDateString("fr-FR"); } catch { return "-"; }
}

function correspondFiltre(statut, filtre) {
    if (filtre === "toutes") return true;
    if (filtre === "attente") return statut === "soumise" || statut === "en_verification";
    return statut === filtre;
}

// ==========================================================
// CHARGEMENT
// ==========================================================

onAuthStateChanged(auth, async (user) => {

    if (!user) {
        window.location.href = "connexion-admin.html";
        return;
    }

    const adminSnap = await getDoc(doc(db, "admins", user.uid));

    if (!adminSnap.exists()) {
        await signOut(auth);
        alert("Accès refusé : ce compte n'a pas les droits administrateur.");
        window.location.href = "connexion-admin.html";
        return;
    }

    await chargerCandidatures();

});

async function chargerCandidatures() {

    document.getElementById("chargement").style.display = "block";

    try {

        const snap = await getDocs(collection(db, "candidaturesRecrutement"));

        candidatures = [];
        snap.forEach((docSnap) => {
            const d = docSnap.data();
            if (d.statut === "brouillon" || !d.statut) return; // dossiers non soumis : ignorés côté admin
            candidatures.push({ id: docSnap.id, ...d });
        });

        mettreAJourStats();
        appliquerFiltres();

    } catch (error) {
        document.getElementById("chargement").textContent = "Erreur lors du chargement : " + error.message;
        return;
    }

    document.getElementById("chargement").style.display = "none";

}

function mettreAJourStats() {
    document.getElementById("nbAttente").textContent = candidatures.filter(c => c.statut === "soumise" || c.statut === "en_verification").length;
    document.getElementById("nbDocValides").textContent = candidatures.filter(c => c.statut === "documents_valides").length;
    document.getElementById("nbTestAutorise").textContent = candidatures.filter(c => c.statut === "test_autorise").length;
    document.getElementById("nbAdmis").textContent = candidatures.filter(c => c.statut === "admis").length;
    document.getElementById("nbRefuses").textContent = candidatures.filter(c => c.statut === "refuse").length;
}

// ==========================================================
// RECHERCHE + FILTRES + TABLEAU
// ==========================================================

function appliquerFiltres() {

    const terme = document.getElementById("recherche").value.trim().toLowerCase();
    const corps = document.getElementById("corpsTableau");
    const messageVide = document.getElementById("messageVide");
    corps.innerHTML = "";

    const resultats = candidatures.filter((c) => {

        if (!correspondFiltre(c.statut, filtreActuel)) return false;
        if (!terme) return true;

        const nom = [c.personnel?.prenom, c.personnel?.nom].filter(Boolean).join(" ").toLowerCase();
        const matieres = (c.matieres || []).join(" ").toLowerCase();
        const ville = (c.personnel?.ville || "").toLowerCase();
        const tel = (c.personnel?.telephone || "").toLowerCase();
        const identifiant = identifiantCourt(c.id).toLowerCase();

        return nom.includes(terme) || matieres.includes(terme) || ville.includes(terme) || tel.includes(terme) || identifiant.includes(terme);

    });

    if (resultats.length === 0) {
        messageVide.style.display = "block";
        return;
    }
    messageVide.style.display = "none";

    resultats.forEach((c) => {

        const nom = [c.personnel?.prenom, c.personnel?.nom].filter(Boolean).join(" ") || "-";
        const initiales = (c.personnel?.nom || "??").substring(0, 2).toUpperCase();
        const photoUrl = c.documents?.photo?.url;
        const statutAffiche = c.statut === "soumise" ? "en_verification" : c.statut;

        const tr = document.createElement("tr");
        tr.innerHTML = `
            <td><div class="avatar-mini">${photoUrl ? `<img src="${photoUrl}" alt="${nom}">` : initiales}</div></td>
            <td>${nom}</td>
            <td>${(c.matieres || []).join(", ") || "-"}</td>
            <td>${c.personnel?.ville || "-"}</td>
            <td>${c.professionnel?.diplome || "-"}</td>
            <td><span class="badge ${statutAffiche}">${LIBELLES_STATUT[c.statut] || c.statut}</span></td>
            <td>${formaterDate(c.dateCreation)}</td>
            <td><button class="btn-voir" data-id="${c.id}">👁️ Voir</button></td>
        `;
        corps.appendChild(tr);

    });

    corps.querySelectorAll(".btn-voir").forEach((btn) => {
        btn.addEventListener("click", () => ouvrirModal(btn.dataset.id));
    });

}

document.getElementById("recherche").addEventListener("input", appliquerFiltres);
document.getElementById("filtreStatut").addEventListener("change", (e) => {
    filtreActuel = e.target.value;
    document.querySelectorAll(".stat-card").forEach(c => c.classList.toggle("actif", c.dataset.filtre === filtreActuel));
    appliquerFiltres();
});

document.querySelectorAll(".stat-card").forEach((carte) => {
    carte.addEventListener("click", () => {
        filtreActuel = carte.dataset.filtre;
        document.getElementById("filtreStatut").value = filtreActuel;
        document.querySelectorAll(".stat-card").forEach(c => c.classList.toggle("actif", c === carte));
        appliquerFiltres();
    });
});

function docCellule(cle, cData) {
    const lien = document.getElementById("lien" + cle.charAt(0).toUpperCase() + cle.slice(1));
    const nomEl = document.getElementById("nomFichier" + cle.charAt(0).toUpperCase() + cle.slice(1));
    const info = cData?.documents?.[cle];
    if (info && info.url) {
        nomEl.textContent = info.nom || "Fichier envoyé";
        lien.innerHTML = `<a href="${info.url}" target="_blank" rel="noopener">${cle === "photo" ? "Voir" : "Ouvrir / Télécharger"}</a>`;
    } else {
        nomEl.textContent = "Aucun fichier envoyé";
        lien.innerHTML = `<span class="manquant">❌ Manquant</span>`;
    }
}

    const c = candidatures.find(x => x.id === id);

    docCellule("photo", c);

    document.getElementById("verifDiplome").checked = !!verif.diplome;
    document.getElementById("verifPieceIdentite").checked = !!verif.pieceIdentite;

    document.getElementById("scoreDossierComplet").value = score.dossierComplet ?? 0;

    document.getElementById("scoreExperience").value = score.experience ?? 0;
    document.getElementById("scorePresentationCv").value = score.presentationCv ?? 0;
    recalculerScore();

    document.getElementById("commentaireAdmin").value = c.messageAdmin || "";

    const listeHisto = document.getElementById("listeHistorique");
    const historique = Array.isArray(c.historique) ? [...c.historique].sort((a, b) => new Date(b.date) - new Date(a.date)) : [];
    listeHisto.innerHTML = historique.length
        ? historique.map(h => `<div class="historique-item"><div class="date">${h.date ? new Date(h.date).toLocaleString("fr-FR") : "-"}</div><div class="action">${h.action}</div></div>`).join("")

    document.querySelectorAll(".onglet-contenu").forEach(o => o.classList.toggle("actif", o.dataset.contenu === "infos"));

    document.getElementById("modalOverlay").classList.add("ouvert");

}

async function afficherTest(id) {

    const zone = document.getElementById("zoneTest");

    try {
        const testSnap = await getDoc(doc(db, "testsRecrutement", id));

        if (!testSnap.exists()) {
            zone.innerHTML = `<div class="historique-vide">Ce candidat n'a pas encore passé le test de recrutement.</div>`;
            return;
        }

        const t = testSnap.data();
        const enCours = !t.terminé;

        const dateAffichee = t.fin ? new Date(t.fin).toLocaleDateString("fr-FR") : (t.debut ? new Date(t.debut).toLocaleDateString("fr-FR") + " (en cours)" : "-");
        const tempsAffiche = t.tempsUtiliseSecondes != null ? Math.round(t.tempsUtiliseSecondes / 60) + " min" : (enCours ? "En cours..." : "-");
        const scoreAffiche = t.score != null ? `${t.score} / 20` : "-";
        const pourcentageAffiche = t.pourcentage != null ? ` (${t.pourcentage} %)` : "";

        zone.innerHTML = `
            <div class="info-ligne"><span class="cle">Statut</span><span class="val">${enCours ? "🟠 En cours" : "✅ Terminé"}</span></div>
            <div class="info-ligne"><span class="cle">Session</span><span class="val">${t.sessionId || "-"}</span></div>
            <div class="info-ligne"><span class="cle">Matière testée</span><span class="val">${t.matiere || "-"}</span></div>
            <div class="info-ligne"><span class="cle">Date</span><span class="val">${dateAffichee}</span></div>
            <div class="info-ligne"><span class="cle">Temps utilisé</span><span class="val">${tempsAffiche}</span></div>
            <div class="info-ligne"><span class="cle">Score</span><span class="val">${scoreAffiche}${pourcentageAffiche}</span></div>
            <div class="info-ligne"><span class="cle">Tentative</span><span class="val">${t.tentativeNumero || 1} / 1</span></div>
            <div class="info-ligne" style="border-bottom:none;"><span class="cle">Changements d'onglet</span><span class="val">${t.changementsOnglet ?? 0}</span></div>
        `;

    } catch (error) {
        zone.innerHTML = `<div class="historique-vide">Erreur lors du chargement du test : ${error.message}</div>`;
    }

}

function recalculerScore() {
    const total = ["scoreDossierComplet", "scoreDiplomesCoherents", "scoreExperience", "scorePresentationCv"]
        .reduce((somme, id) => somme + (Number(document.getElementById(id).value) || 0), 0);
    document.getElementById("scoreTotalAffiche").textContent = total;
    return total;
}

function recupererVerification() {
    return {
        cv: document.getElementById("verifCv").checked,

function recupererScore() {
    return {
        dossierComplet: Number(document.getElementById("scoreDossierComplet").value) || 0,
        diplomesCoherents: Number(document.getElementById("scoreDiplomesCoherents").value) || 0,
        experience: Number(document.getElementById("scoreExperience").value) || 0,

    };
}

// Envoie un vrai message dans la messagerie du candidat (visible sur messages-enseignant.html)
async function envoyerNotification(candidature, contenu) {
    try {
        const profilSnap = await getDoc(doc(db, "enseignants", candidature.id));
        const identifiantEnseignant = profilSnap.exists() ? profilSnap.data().identifiant : null;
        if (!identifiantEnseignant) return;

        await addDoc(collection(db, "messages"), {
            enseignantId: identifiantEnseignant,
            contenu,
            dateEnvoi: new Date().toISOString(),
            origine: "recrutement"
        });
    } catch (error) {
        console.warn("Notification non envoyée :", error.message);
    }
}

async function appliquerAction({ nouveauStatut, nouveauDocumentsStatut, actionHistorique, messageParDefaut, exigerCommentaire }) {

    const c = candidatures.find(x => x.id === candidatureOuverteId);
    if (!c) return;

    const commentaire = document.getElementById("commentaireAdmin").value.trim();

    if (exigerCommentaire && !commentaire) {
        window.notify("Merci d'ajouter un commentaire avant cette action.", "avertissement");
        return;
    }

    window.showLoading("Mise à jour du dossier...");

    try {

        const maj = {
            verificationDocs: recupererVerification(),
            scoreAdmin: recupererScore(),
            messageAdmin: commentaire || messageParDefaut,
            historique: arrayUnion({ date: new Date().toISOString(), action: actionHistorique })
        };

        if (nouveauStatut) maj.statut = nouveauStatut;
        if (nouveauDocumentsStatut) maj.documentsStatut = nouveauDocumentsStatut;

        await updateDoc(doc(db, "candidaturesRecrutement", candidatureOuverteId), maj);

        await envoyerNotification(c, commentaire || messageParDefaut);

        window.hideLoading();
        window.notify("Dossier mis à jour avec succès.", "succes");

        document.getElementById("modalOverlay").classList.remove("ouvert");
        await chargerCandidatures();

    } catch (error) {
        window.hideLoading();
        window.notify("Erreur : " + error.message, "erreur");
    }

}
